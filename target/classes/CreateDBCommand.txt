create database hhr;

create table project
(id int not null primary key auto_increment,
client varchar (100) not null,
card_status int,
FOREIGN KEY (card_status) REFERENCES cardstatus (Id),
project_type int,
FOREIGN KEY (project_type) REFERENCES projecttype (Id),
id_employee int not null,
name varchar (200) not null,
func_area text,
subject_area text,
description text not null,
tasks text not null,
stage int,
FOREIGN KEY (stage) REFERENCES projectstage (Id),
dead_line date not null,
technoligies text not null,
dev_method int,
FOREIGN KEY (dev_method) REFERENCES devmethodology (Id),
stake_holders int,
product boolean not null,
analitics int not null,
devs int not null,
testers boolean not null,
tech_writers boolean not null,
team int not null,
location text,
timeplan int,
FOREIGN KEY (timeplan) REFERENCES timeplan (Id),
deadline_hr date not null,
overtimes int,
FOREIGN KEY (overtimes) REFERENCES overtimes (Id),
hr_details text not null,
gost boolean not null);



create table devmethodology (id int not null primary key auto_increment, name varchar (200) not null);
create table overtimes (id int not null primary key auto_increment, name varchar (200) not null);
create table cardstatus (id int not null primary key auto_increment, name varchar (200) not null);
create table projectstage (id int not null primary key auto_increment, name varchar (200) not null);
create table projecttype (id int not null primary key auto_increment, name varchar (200) not null);
create table timeplan (id int not null primary key auto_increment, name varchar (200) not null);
